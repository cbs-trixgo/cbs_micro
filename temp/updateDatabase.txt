// UPDATE amountChilds CHO DANH Báº  done

let listContact = db.contacts.find({}).toArray();
listContact.forEach(item => {
    let amountChilds = item.childs?.length || 0;
    db.contacts.updateOne({ _id: ObjectId(item._id)}, { $set: { amountChilds }});
})
